```C++
#include <iostream>

/* run this program using the console pauser or add your own getch, system("pause") or input loop */
using namespace std;

int main(int argc, char** argv) {
	int value;
	int a;
	while(true)
	{	
		cout << "請輸入數字：";
		cin >> value;
		if(value==0)
		{
			a = 0;
			cout << "已清空" << endl;
			value=0;	
			
		}
		else
		{	
			a += value;
			cout << "目前累積:" << a <<endl;	
		}
			
	}
	
	
	return 0;
}
```
